The Core Algorithm

Bridging the digital pixel space to physical coordinates via Affine Transformation.

import numpy as np

# Hand-Eye Calibration Matrix
def pixel_to_physical(u, v, M):
    """
    Convert (u,v) pixel coords to (x,y)
    using Affine Transformation Matrix M
    """
    pixel_vector = np.array([u, v, 1])
    
    # M dot Product
    physical_vector = M @ pixel_vector
    
    x = physical_vector[0]
    y = physical_vector[1]
    
    return (x, y)
